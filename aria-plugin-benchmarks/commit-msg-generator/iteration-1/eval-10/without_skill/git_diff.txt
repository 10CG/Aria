+"""Unit tests for date formatting utilities."""
+
+import pytest
+from datetime import datetime, timezone
+
+from src.utils.date_utils import format_date, to_timezone
+
+
+class TestDateFormatting:
+    """Test date formatting functionality."""
+    
+    def test_format_iso_date(self):
+        """Test ISO 8601 date formatting."""
+        dt = datetime(2024, 1, 15, 14, 30, 0)
+        result = format_date(dt, format='iso')
+        assert result == '2024-01-15T14:30:00'
+    
+    def test_format_readable_date(self):
+        """Test human-readable date formatting."""
+        dt = datetime(2024, 1, 15, 14, 30, 0)
+        result = format_date(dt, format='readable')
+        assert 'Jan' in result and '15' in result
+    
+    def test_format_custom_pattern(self):
+        """Test custom format pattern."""
+        dt = datetime(2024, 1, 15, 14, 30, 0)
+        result = format_date(dt, format='%Y/%m/%d')
+        assert result == '2024/01/15'
+
+
+class TestTimezoneConversion:
+    """Test timezone conversion functionality."""
+    
+    def test_utc_to_est(self):
+        """Test UTC to EST conversion."""
+        dt = datetime(2024, 1, 15, 14, 30, 0, tzinfo=timezone.utc)
+        result = to_timezone(dt, 'America/New_York')
+        assert result.tzinfo.zone == 'America/New_York'
+    
+    def test_none_input(self):
+        """Test handling of None input."""
+        result = to_timezone(None, 'America/New_York')
+        assert result is None
