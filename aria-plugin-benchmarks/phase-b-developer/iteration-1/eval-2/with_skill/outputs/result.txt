```python
# backend/api/auth/github_oauth.py
import requests
from backend.services.auth_service import create_or_link_user
from backend.utils.jwt_handler import generate_jwt
from backend.utils.csrf import validate_state_token


def exchange_code(code: str) -> dict:
    response = requests.post('https://github.com/login/oauth/access_token', ...)
    return response.json()


def github_callback(code: str, state: str) -> dict:
    if not validate_state_token(state):
        raise ValueError('Invalid state token')

    token_data = exchange_code(code)
    github_user = get_github_user_profile(token_data['access_token'])
    user, created = create_or_link_user(provider='github', profile=github_user)

    return {
        'jwt_token': generate_jwt(user.id),
        'user_created': created,
        'user_id': user.id
    }
```
